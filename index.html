# ATOM — Voice-controlled AI Assistant with Arduino Integration
# Author: Snuhith | 2026

import os, re, glob, random, signal, serial, tempfile
import pyttsx3, pygame, PyPDF2, requests
import datetime as dt
import speech_recognition as sr

VA_NAME = "atom"
ARDUINO_PORT = "COM6"
WEATHER_API_KEY = "your_key_here"

ARDUINO_COMMANDS = {
    "shake_hand": b"9", "walk": b"8", "back": b"7",
    "left": b"6",  "right": b"5", "led_on": b"0", "led_off": b"1",
}

class VoiceAssistant:
    def __init__(self):
        self.listener = sr.Recognizer()
        self.listener.energy_threshold = 100
        self.arduino = None
        try:
            self.arduino = serial.Serial(ARDUINO_PORT, 9600)
        except Exception as e:
            print(f"[WARN] Arduino not connected: {e}")
        self.running = True

    def speak(self, text):
        print(f"[ATOM] {text}")
        engine = pyttsx3.init()
        engine.setProperty("rate", 150)
        engine.say(text)
        engine.runAndWait()

    def listen(self):
        try:
            with sr.Microphone() as source:
                voice = self.listener.listen(source, timeout=8)
                return self.listener.recognize_google(voice).lower()
        except:
            return ""

    def handle_command(self, cmd):
        if "weather in" in cmd:
            city = cmd.split("weather in", 1)[1].strip()
            self.speak(self.get_weather(city))
        elif "time" in cmd:
            self.speak(f"Time is {dt.datetime.now().strftime('%I:%M %p')}")
        elif "play" in cmd:
            self.speak("Playing music...")
        elif "walk" in cmd:
            self.arduino.write(b"8")
            self.speak("Walking forward.")
        elif "shake hand" in cmd:
            self.arduino.write(b"9")
            self.speak("Shaking hand!")
        elif cmd in ("stop", "quit"):
            self.speak("Goodbye!")
            self.running = False

    def run(self):
        self.speak(f"Hello! I am {VA_NAME}. How can I help you?")
        while self.running:
            cmd = self.listen()
            self.handle_command(cmd)

if __name__ == "__main__":
    VoiceAssistant().run()
